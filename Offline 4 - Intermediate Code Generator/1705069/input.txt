int max2(int a, int b){
	if (a >= b) {
		return a;
	} else {
		return b;
	}
}

int max3(int a, int b, int c){
	int maxab, maxabc;
	maxab = max2(a, b);
	return max2(c, maxab);
}

int factorial(int n){
	int i, fact;
	i = 1;
	fact = i;
	for(i = 1; i <= n; i++){
		fact = fact * i;
	}
	return fact;
}
